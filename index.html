1. Class & Object (Tanpa Constructor)


class Car {
  String? model;
  String? color;

  void start() {
    print('Car started!');
  }

  void stop() {
    print('Car stopped!');
  }
}

void main() {
  Car myCar = Car();
  myCar.model = "Sedan";
  myCar.color = "Merah";
  myCar.start();
  myCar.stop();
}

2. Constructor


class Car {
  String? model;
  String? color;

  Car(this.model, this.color);

  void start() {
    print('Car started!');
  }
}

void main() {
  Car myCar = Car('Toyota', 'Red');
  print('model car: ${myCar.model}, color: ${myCar.color}');
  myCar.start();
}

3. Validasi Constructor (Mahasiswa)


class Mahasiswa {
  String nama;
  String nim;
  double ipk;

  Mahasiswa(this.nama, this.nim, this.ipk) {
    if (ipk < 0.0 || ipk > 4.0) {
      throw ArgumentError('IPK harus berada dalam rentang 0.0 - 4.0');
    }
  }
}

4. Encapsulation


class Person {
  String _name;
  int age;

  Person(this._name, this.age);

  void setName(String name) {
    _name = name;
  }

  String getName() {
    return _name;
  }
}

5. Named Parameter Constructor


class Produk {
  final String id;
  final String nama;
  final double harga;
  final int stok;
  final String kategori;

  Produk({
    required this.id,
    required this.nama,
    required this.harga,
    this.stok = 0,
    this.kategori = 'Umum',
  });
}

Kesimpulan

OOP mempermudah pengelolaan program karena struktur kode menjadi lebih rapi, mudah dikembangkan, serta lebih aman melalui konsep enkapsulasi. Konsep seperti constructor dan named parameter membantu dalam inisialisasi data.
